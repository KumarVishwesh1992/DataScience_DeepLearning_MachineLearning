def corrank(X):
    import itertools
    data = pd.DataFrame([[i,j,X.corr().loc[i,j]] for i,j in list(itertools.combinations(X.corr(), 2))],
                      columns=['pair-1','pair-2','corr'])    
    #print(df.sort_values(by='corr',ascending=False))
    return data
